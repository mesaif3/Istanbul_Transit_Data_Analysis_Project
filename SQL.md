- Dates

```sql
-- column name is based on the date_col_name parameter
-- scans all the loaded dates accross all the available tables
WITH all_dates AS (
    SELECT DISTINCT transition_date FROM table_1
    UNION
    SELECT DISTINCT transition_date FROM table_2
    UNION
    ...
    SELECT DISTINCT transition_date FROM table_n
),
-- a column table with only distinct dates
WITH scanned_dates AS (
    SELECT DISTINCT transition_date::DATE AS transition_date
    FROM (SELECT * FROM all_dates)
    ORDER BY transition_date
)
```

```sql
-- column name is based on the date_col_name parameter
-- a table containing the earliest date from each table
WITH first_dates AS (
    SELECT DISTINCT transition_date FROM table_1
    ORDER BY transition_date ASC LIMIT 1
    UNION
    ...
    SELECT DISTINCT transition_date FROM table_n
    ORDER BY transition_date ASC LIMIT 1
),
-- a table containing the latest date from each table
WITH last_dates AS (
    SELECT DISTINCT transition_date FROM table_1
    ORDER BY transition_date DESC LIMIT 1
    UNION
    ...
    SELECT DISTINCT transition_date FROM table_n
    ORDER BY transition_date DESC LIMIT 1
),
-- a table containing a range of dates from earliest to the latest
WITH generated_dates AS (
    SELECT CAST(range AS DATE) AS transition_date
        FROM range(
            (SELECT * FROM table_first_date LIMIT 1),
            (SELECT * FROM table_last_date LIMIT 1) + INTERVAL 1 DAY,
            INTERVAL 1 DAY
            )
)
```

-- extract the date info
WITH table_date AS (
    SELECT
        -- column name is based on the date_col_name parameter
        transition_date, -- 01/01/2023
        CAST(date_part('year', transition_date) AS INT) AS Year,    -- 2023
        'Q' || CAST(date_part('quarter', transition_date) AS INT) AS Quarter, -- Q1
        strftime(transition_date, '%B') AS Month_Name, -- January
        CAST(date_part('month', transition_date) AS INT) AS Month_No, -- 1
        CAST(date_part('day', transition_date) AS INT) AS Day, -- 1
        strftime(transition_date, '%A') AS Weekday_Name, -- Sunday
        CAST(date_part('isodow', transition_date) AS INT) AS Weekday_No, -- 7
    FROM table_dates
)
-- create a new, up-to-date, dates table in my database
CREATE OR REPLACE TABLE 'dates' AS SELECT * FROM table_date
